#include<stdio.h>
int hedong(int x)
{
	int i,s[3],p,j,t,sum1,sum2,d;
	int w[100];
	w[0]=x;
	w[1]=1;
	for(d=0;w[d]!=w[d-1];d++)
	{
		s[0]=w[d]/100;
		s[1]=(w[d]/10)%10;
		s[2]=w[d]%10;
		for(i=0;i<2;i++)
			for(j=0;j<2-i;j++)
				if(s[j]>s[j+1])
				{
					t=s[j];
					s[j]=s[j+1];
					s[j+1]=t;
				}
				sum1=s[0]+s[1]*10+s[2]*100;
				sum2=s[2]+s[1]*10+s[0]*100;
				p=sum1-sum2;
				if(p>0)
					w[d+1]=p;
				else
					w[d+1]=p*(-1);
	}
	return d;
}
