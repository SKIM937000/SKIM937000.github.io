 写一函数，使输入的一个字符串按反序存放，在主函数中输入和输出字符串。
输入
abc
输出
cba 

```c
#include<stdio.h>
#include<string.h>
void reverse (char s[20])
{
	int i;
	char a;
	int t;
t=strlen(s);/*
for(i=0;s[i]!='\0';i++)
{t++};或者
for(i=0;s[i]!='\0',i++)
{t=i+1;}
*/
    //在for循环中，循环体循环后，不满足条件，第三条指令执行，break
	for(i=t;i>t/2;i--)
	{	
		a=s[t-i];
	s[t-i]=s[i-1];
	s[i-1]=a;
	}

}
```
