#include<stdio.h>
#include <math.h>
int prime(int n)
{
    int i;
    for(i = 2; i <= sqrt(n); i ++)
    {
        if(n%i==0) return 0;
    }
    return 1;
}
int main() 
{
    int m, i;
    scanf("%d",&m);

for(i = 2; i < m; i ++)
    if(prime(i) && prime(m-i))
    {
        printf("%d=%d+%d\n", m,i,m-i);
        break;
    }
return 0;



}
